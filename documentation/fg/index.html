Methods

.init()
  • initializes the framework 

.update(dt[number])
  • dt: delta value passed from the main loop 

.draw()
  • draws the engine 

.run()
  • fuccboiGDX’s main loop, override love.run with it like this:

function love.run()
    math.randomseed(os.time())
    math.random() math.random()
    if love.math then love.math.setRandomSeed(os.time()) end
    if love.event then love.event.pump() end
    if love.load then love.load(arg) end
    if love.timer then love.timer.step() end
    fg.run()
end



.Class()
  • creates an engine class 

.getPS(name[string])
  • returns the built particle system from file fuccboi/resources/particles/sperm/name.lua 

.getUID()
  • number returns a new unique identifier 

.keypressed(key[string])
  • alias for fg.input:keypressed(key) 

.keyreleased(key[string])
  • alias for fg.input:keyreleased(key) 

.mousepressed(button[string])
  • alias for fg.input:mousepressed(button) 

.mousereleased(button[string])
  • alias for fg.input:mousereleased(button) 

.gamepadpressed(joystick[Joystick], button[string])
  • alias for fg.input:gamepadpressed(joystick, button) 

.gamepadreleased(joystick[Joystick], button[string])
  • alias for fg.input:gamepadreleased(joystick, button) 

.gamepadaxis(joystick[Joystick], axis[GamepadAxis], new_value[number])
  • alias for fg.input:gamepadaxis(joystick, axis, new_value) 

.textinput(text[string])
  • alias for fg.loveframes.textinput(text) 

.resize(w[number], h[number])
  • resizes the screen to the new w, h values 

.setScreenSize(w[number], h[number])
  • changes the screen size to the new w, h values

Attributes

.Assets[table]
  • a table to be used with Loader to hold resources 

.classes[table]
  • a table that holds all classes created with the fg.Class call 

.debug_draw[boolean]
  • if debug drawing is enabled or not 

.lovebird_enabled[boolean]
  • if lovebird is enabled or not 

.min_width[number], .min_height[number]
  • the minimum width and height the screen can be resized to 

.screen_scale[number]
  • the amount of scaling (zoom) added to the screen 

.screen_width[number], .screen_height[]
  • the screen width and height 

.Shaders[table]
  • a table to hold shaders 

.Spritebatches[table]
  • a table to hold spritebatches
